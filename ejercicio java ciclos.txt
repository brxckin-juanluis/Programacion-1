                    System.out.print("humano ingresa Un Numero : ");

                    }
                    break;
                case 2:

                    String respuesta = scanner.next().toLowerCase();
                    switch (respuesta) {
                        case "si":
                            System.out.println("bienvenido");

                        case "no":
                            System.out.println("adios");
                            break;
                        default:
                            break;
                    }
                    break;
                case 3:
                    System.out.println("Ingresa el primer angulo");
                    Integer anguloUno = scanner.nextInt();
                    System.out.println("Ingresa el segundo angulo");

                    Integer t = 180 - (anguloUno + anguloDos);
                    if (t < 0) {

                default:
