def elevar_potencia(base, exponente):
    resultado = base ** exponente
    return resultado

def factorial(numero):
    if numero == 0:
        return 1
    else:
        return numero * factorial(numero - 1)

def main():
    try:
        numero = int(input("humano ingresa un número para encontrar su factorial: "))
        if numero < 0:
            print("El factorial no está definido para números negativos humano.")
        else:
            resultado = factorial(numero)
            print(f"El factorial de {numero} es: {resultado}")
    except ValueError:
        print("Por favor humano, ingresa un número entero válido.")

if __name__ == "__main__":
    main()
